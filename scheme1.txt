(define (retangulo-maior lado-a1 lado-a2 lado-b1 lado-b2)
		(define A (* lado-a2 lado-a1))
		(define B (* lado-b1 lado-b2))
		(define dif (- A B))
		(display "Retangulo A: ") (display A) (newline)
		(display "Retangulo B: ") (display B) (newline)
		(if ( > dif 0) (print "O retangulo A é maior " dif " unidades")
		    (if ( < dif 0) (print "O retangulo B é maior " (- dif) " unidades")
		        (print "Os retangulos apresentam areas iguais")))
     )
